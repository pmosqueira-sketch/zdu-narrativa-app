  <script type="module">

    import { createClient } from 'https://cdn.jsdelivr.net/npm/@supabase/supabase-js/+esm';


    const supabaseUrl = 'https://lilxvmvykdssotolctbl.supabase.co';

    const supabaseKey = 'sb_publishable_QdAzhd79yASprzQ1q5VHhQ_64RL1cAW';

    const supabase = createClient(supabaseUrl, supabaseKey);


    async function fetchCampañas() {

      const { data, error } = await supabase

        .from('campañas')

        .select('*')

        .eq('estatus', 'activa');


      const container = document.getElementById('campañas');


      if (error) {

        console.error('Error al obtener campañas:', error);

        container.innerHTML = '<p>Error al cargar campañas.</p>';

        return;

      }


      if (!data || data.length === 0) {

        container.innerHTML = '<p>No hay campañas activas por el momento.</p>';

        return;

      }


      const html = data.map(c => `

        <div class="campana">

          <h3>${c.nombre}</h3>

          <p>${c.descripcion}</p>

        </div>

      `).join('');


      container.innerHTML = html;

    }


    fetchCampañas();

  </script>
